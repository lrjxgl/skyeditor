# 配置方法
    //编辑器位置
    skyeditor.root = "http://" + document.domain + "/plugin/skyeditor/";
    //图片上传地址
    skyeditor.uploadUrl = "/shop.php?m=upload&a=uploadTao";
    //视频上传地址
    skyeditor.uploadVideo="/shop.php?m=upload&a=UploadVideo";
